72 def _ParseAndStripGTestFlags(argv):
73   """Parses and strips Google Test flags from argv.  This is idempotent."""
74 
75   # Suppresses the lint complaint about a global variable since we need it
76   # here to maintain module-wide state.
77   global _gtest_flags_are_parsed  # pylint: disable-msg=W0603
78   if _gtest_flags_are_parsed:
79     return
80 
81   _gtest_flags_are_parsed = True
82   for flag in _flag_map:
83     # The environment variable overrides the default value.
84     if flag.upper() in os.environ:
85       _flag_map[flag] = os.environ[flag.upper()]
86 
87     # The command line flag overrides the environment variable.
88     i = 1  # Skips the program name.
89     while i < len(argv):
90       prefix = '--' + flag + '='
91       if argv[i].startswith(prefix):
92         _flag_map[flag] = argv[i][len(prefix):]
93         del argv[i]
94         break
95       else:
96         # We don't increment i in case we just found a --gtest_* flag
97         # and removed it from argv.
98         i += 1

173 def GetExitStatus(exit_code):
174   """Returns the argument to exit(), or -1 if exit() wasn't called.
175 
176   Args:
177     exit_code: the result value of os.system(command).
178   """
179 
180   if os.name == 'nt':
181     # On Windows, os.WEXITSTATUS() doesn't work and os.system() returns
182     # the argument to exit() directly.
183     return exit_code
184   else:
185     # On Unix, os.WEXITSTATUS() must be used to extract the exit status
186     # from the result of os.system().
187     if os.WIFEXITED(exit_code):
188       return os.WEXITSTATUS(exit_code)
189     else:
190       return -1
191

143   """Returns the absolute path of the test binary given its name.
144 
145   The function will print a message and abort the program if the resulting file
146   doesn't exist.
147 
148   Args:
149     executable_name: name of the test binary that the test script runs.
150     build_dir:       directory where to look for executables, by default
151                      the result of GetBuildDir().
152 
153   Returns:
154     The absolute path of the test binary.
155   """
156 
157   path = os.path.abspath(os.path.join(build_dir or GetBuildDir(),
158                                       executable_name))
159   if (IS_WINDOWS or IS_CYGWIN) and not path.endswith('.exe'):
160     path += '.exe'
161 
162   if not os.path.exists(path):
163     message = (
164         'Unable to find the test binary. Please make sure to provide path\n'
165         'to the binary via the --build_dir flag or the BUILD_DIR\n'
166         'environment variable.')
167     print >> sys.stderr, message
168     sys.exit(1)
169
