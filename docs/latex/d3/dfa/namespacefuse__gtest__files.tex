176 def FuseGTestAllCcToFile(gtest_root, output_file):
177   """Scans folder gtest_root to generate gtest/gtest-all.cc in output_file."""
178 
179   processed_files = sets.Set()
180 
181   def ProcessFile(gtest_source_file):
182     """Processes the given gtest source file."""
183 
184     # We don't process the same #included file twice.
185     if gtest_source_file in processed_files:
186       return
187 
188     processed_files.add(gtest_source_file)
189 
190     # Reads each line in the given gtest source file.
191     for line in file(os.path.join(gtest_root, gtest_source_file), 'r'):
192       m = INCLUDE_GTEST_FILE_REGEX.match(line)
193       if m:
194         if 'include/' + m.group(1) == GTEST_SPI_H_SEED:
195           # It's '#include "gtest/gtest-spi.h"'.  This file is not
196           # #included by "gtest/gtest.h", so we need to process it.
197           ProcessFile(GTEST_SPI_H_SEED)
198         else:
199           # It's '#include "gtest/foo.h"' where foo is not gtest-spi.
200           # We treat it as '#include "gtest/gtest.h"', as all other
201           # gtest headers are being fused into gtest.h and cannot be
202           # #included directly.
203 
204           # There is no need to #include "gtest/gtest.h" more than once.
205           if not GTEST_H_SEED in processed_files:
206             processed_files.add(GTEST_H_SEED)
207             output_file.write('#include "%s"\n' % (GTEST_H_OUTPUT,))
208       else:
209         m = INCLUDE_SRC_FILE_REGEX.match(line)
210         if m:
211           # It's '#include "src/foo"' - let's process it recursively.
212           ProcessFile(m.group(1))
213         else:
214           output_file.write(line)
215 
216   ProcessFile(GTEST_ALL_CC_SEED)

112   """Verifies that the given output file path is valid.

115   """
116 
117   # Makes sure the output file either doesn't exist or can be overwritten.
118   output_file = os.path.join(output_dir, relative_path)
119   if os.path.exists(output_file):
120     # TODO(wan@google.com): The following user-interaction doesn't

128       sys.exit(1)
