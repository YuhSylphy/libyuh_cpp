00534 // An assertion predicate determining whether a one string is a prefix for
00535 // another.
00536 template <typename StringType>
00537 AssertionResult HasPrefix(const StringType& str, const StringType& prefix) {
00538   if (str.find(prefix, 0) == 0)
00539     return AssertionSuccess();
00540 
00541   const bool is_wide_string = sizeof(prefix[0]) > 1;
00542   const char* const begin_string_quote = is_wide_string ? "L\"" : "\"";
00543   return AssertionFailure()
00544       << begin_string_quote << prefix << "\" is not a prefix of "
00545       << begin_string_quote << str << "\"\n";
00546 }
