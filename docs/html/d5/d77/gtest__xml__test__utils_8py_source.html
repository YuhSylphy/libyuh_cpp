00052     """
00053     Asserts that actual_node (a DOM node object) is equivalent to
00054     expected_node (another DOM node object), in that either both of
00055     them are CDATA nodes and have the same value, or both are DOM
00056     elements and actual_node meets all of the following conditions:
00057 
00058     *  It has the same tag name as expected_node.
00059     *  It has the same set of attributes as expected_node, each with
00060        the same value as the corresponding attribute of expected_node.
00061        Exceptions are any attribute named "time", which needs only be
00062        convertible to a floating-point number and any attribute named
00063        "type_param" which only has to be non-empty.
00064     *  It has an equivalent set of child nodes (including elements and
00065        CDATA sections) as expected_node.  Note that we ignore the
00066        order of the children as they are not guaranteed to be in any
00067        particular order.
00068     """
00069 
00070     if expected_node.nodeType == Node.CDATA_SECTION_NODE:
00071       self.assertEquals(Node.CDATA_SECTION_NODE, actual_node.nodeType)
00072       self.assertEquals(expected_node.nodeValue, actual_node.nodeValue)
00073       return

00076     self.assertEquals(Node.ELEMENT_NODE, expected_node.nodeType)
00077     self.assertEquals(expected_node.tagName, actual_node.tagName)
00078 
00079     expected_attributes = expected_node.attributes
00080     actual_attributes   = actual_node  .attributes

00082         expected_attributes.length, actual_attributes.length,

00113   def _GetChildren(self, element):
00114     """
00115     Fetches all of the child nodes of element, a DOM Element object.
00116     Returns them as the values of a dictionary keyed by the IDs of the
00117     children.  For <testsuites>, <testsuite> and <testcase> elements, the ID
00118     is the value of their "name" attribute; for <failure> elements, it is
00119     the value of the "message" attribute; CDATA sections and non-whitespace
00120     text nodes are concatenated into a single CDATA section with ID
00121     "detail".  An exception is raised if any element other than the above
00122     four is encountered, if two child elements with the same identifying
00123     attributes are encountered, or if any other type of node is encountered.
00124     """
00125 
00126     children = {}
00127     for child in element.childNodes:
00128       if child.nodeType == Node.ELEMENT_NODE:
00129         self.assert_(child.tagName in self.identifying_attribute,
00130                      "Encountered unknown element <%s>" % child.tagName)
00131         childID = child.getAttribute(self.identifying_attribute[child.tagName])
00132         self.assert_(childID not in children)
00133         children[childID] = child
00134       elif child.nodeType in [Node.TEXT_NODE, Node.CDATA_SECTION_NODE]:
00135         if "detail" not in children:
00136           if (child.nodeType == Node.CDATA_SECTION_NODE or
00137               not child.nodeValue.isspace()):
00138             children["detail"] = child.ownerDocument.createCDATASection(
00139                 child.nodeValue)
00140         else:
00141           children["detail"].nodeValue += child.nodeValue
00142       else:
00143         self.fail("Encountered unexpected node type %d" % child.nodeType)
00144     return children

00146   def NormalizeXml(self, element):
00147     """
00148     Normalizes Google Test's XML output to eliminate references to transient
00149     information that may change from run to run.
00150 
00151     *  The "time" attribute of <testsuites>, <testsuite> and <testcase>
00152        elements is replaced with a single asterisk, if it contains
00153        only digit characters.
00154     *  The "type_param" attribute of <testcase> elements is replaced with a
00155        single asterisk (if it sn non-empty) as it is the type name returned
00156        by the compiler and is platform dependent.
00157     *  The line number reported in the first line of the "message"
00158        attribute of <failure> elements is replaced with a single asterisk.
00159     *  The directory names in file paths are removed.
00160     *  The stack traces are removed.
00161     """
00162 
00163     if element.tagName in ("testsuites", "testsuite", "testcase"):
00164       time = element.getAttributeNode("time")
00165       time.value = re.sub(r"^\d+(\.\d+)?$", "*", time.value)
00166       type_param = element.getAttributeNode("type_param")
00167       if type_param and type_param.value:
00168         type_param.value = "*"
00169     elif element.tagName == "failure":
00170       for child in element.childNodes:
00171         if child.nodeType == Node.CDATA_SECTION_NODE:
00172           # Removes the source line number.
00173           cdata = re.sub(r"^.*[/\\](.*:)\d+\n", "\\1*\n", child.nodeValue)
00174           # Removes the actual stack trace.
00175           child.nodeValue = re.sub(r"\nStack trace:\n(.|\n)*",
00176                                    "", cdata)
00177     for child in element.childNodes:
