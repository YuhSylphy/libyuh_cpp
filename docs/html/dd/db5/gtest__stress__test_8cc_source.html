00088 // Calls a large number of Google Test assertions, where exactly one of them
00089 // will fail.
00090 void ManyAsserts(int id) {
00091   GTEST_LOG_(INFO) << "Thread #" << id << " running...";
00092 
00093   SCOPED_TRACE(Message() << "Thread #" << id);
00094 
00095   for (int i = 0; i < kThreadCount; i++) {
00096     SCOPED_TRACE(Message() << "Iteration #" << i);
00097 
00098     // A bunch of assertions that should succeed.
00099     EXPECT_TRUE(true);
00100     ASSERT_FALSE(false) << "This shouldn't fail.";
00101     EXPECT_STREQ("a", "a");
00102     ASSERT_LE(5, 6);
00103     EXPECT_EQ(i, i) << "This shouldn't fail.";
00104 
00105     // RecordProperty() should interact safely with other threads as well.
00106     // The shared_key forces property updates.
00107     Test::RecordProperty(IdToKey(id, "string").c_str(), IdToString(id).c_str());
00108     Test::RecordProperty(IdToKey(id, "int").c_str(), id);
00109     Test::RecordProperty("shared_key", IdToString(id).c_str());
00110 
00111     // This assertion should fail kThreadCount times per thread.  It
00112     // is for testing whether Google Test can handle failed assertions in a
00113     // multi-threaded context.
00114     EXPECT_LT(i, 0) << "This should always fail.";
00115   }
00116 }

00126 // Tests using SCOPED_TRACE() and Google Test assertions in many threads
00127 // concurrently.
00128 TEST(StressTest, CanUseScopedTraceAndAssertionsInManyThreads) {
00129   {
00130     scoped_ptr<ThreadWithParam<int> > threads[kThreadCount];
00131     Notification threads_can_start;
00132     for (int i = 0; i != kThreadCount; i++)
00133       threads[i].reset(new ThreadWithParam<int>(&ManyAsserts,
00134                                                 i,
00135                                                 &threads_can_start));
00136 
00137     threads_can_start.Notify();
00138 
00139     // Blocks until all the threads are done.
00140     for (int i = 0; i != kThreadCount; i++)
00141       threads[i]->Join();
00142   }
00143 
00144   // Ensures that kThreadCount*kThreadCount failures have been reported.
00145   const TestInfo* const info = UnitTest::GetInstance()->current_test_info();
00146   const TestResult* const result = info->result();
00147 
00148   std::vector<TestProperty> properties;
00149   // We have no access to the TestResult's list of properties but we can
00150   // copy them one by one.
00151   for (int i = 0; i < result->test_property_count(); ++i)
00152     properties.push_back(result->GetTestProperty(i));
00153 
00154   EXPECT_EQ(kThreadCount * 2 + 1, result->test_property_count())
00155       << "String and int values recorded on each thread, "
00156       << "as well as one shared_key";
00157   for (int i = 0; i < kThreadCount; ++i) {
00158     ExpectKeyAndValueWereRecordedForId(properties, i, "string");
00159     ExpectKeyAndValueWereRecordedForId(properties, i, "int");
00160   }
00161   CheckTestFailureCount(kThreadCount*kThreadCount);
00162 }
