00281   pthread_attr_t  attr;
00282   pthread_t       thread_id;
00283 
00284   // TODO(vladl@google.com): turn mutex into internal::Mutex for automatic
00285   // destruction.
00286   pthread_mutex_init(&mutex, NULL);
00287   pthread_mutex_lock(&mutex);
00288   ASSERT_EQ(0, pthread_attr_init(&attr));
00289   ASSERT_EQ(0, pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE));
00290 
00291   const int status = pthread_create(&thread_id, &attr, &ThreadFunc, &mutex);
00292   ASSERT_EQ(0, pthread_attr_destroy(&attr));
00293   ASSERT_EQ(0, status);
00294   EXPECT_EQ(2U, GetThreadCount());
00295   pthread_mutex_unlock(&mutex);
00296 
00297   void* dummy;
00298   ASSERT_EQ(0, pthread_join(thread_id, &dummy));
00299 
00300   // MacOS X may not immediately report the updated thread count after
00301   // joining a thread, causing flakiness in this test. To counter that, we
00302   // wait for up to .5 seconds for the OS to report the correct value.
00303   for (int i = 0; i < 5; ++i) {
00304     if (GetThreadCount() == 1)
00305       break;

00307     SleepMilliseconds(100);
00308   }
00309   EXPECT_EQ(1U, GetThreadCount());

01013 class AtomicCounterWithMutex {
01014  public:
01015   explicit AtomicCounterWithMutex(Mutex* mutex) :
01016     value_(0), mutex_(mutex), random_(42) {}
01017 
01018   void Increment() {
01019     MutexLock lock(mutex_);
01020     int temp = value_;
01021     {
01022       // Locking a mutex puts up a memory barrier, preventing reads and
01023       // writes to value_ rearranged when observed from other threads.
01024       //
01025       // We cannot use Mutex and MutexLock here or rely on their memory
01026       // barrier functionality as we are testing them here.
01027       pthread_mutex_t memory_barrier_mutex;
01028       GTEST_CHECK_POSIX_SUCCESS_(
01029           pthread_mutex_init(&memory_barrier_mutex, NULL));
01030       GTEST_CHECK_POSIX_SUCCESS_(pthread_mutex_lock(&memory_barrier_mutex));
01031 
01032       SleepMilliseconds(random_.Generate(30));
01033 
01034       GTEST_CHECK_POSIX_SUCCESS_(pthread_mutex_unlock(&memory_barrier_mutex));
01035     }
01036     value_ = temp + 1;
01037   }
01038   int value() const { return value_; }
01039 
01040  private:
01041   volatile int value_;
01042   Mutex* const mutex_;  // Protects value_.
01043   Random       random_;
01044 };
01045 
01046 void CountingThreadFunc(pair<AtomicCounterWithMutex*, int> param) {
01047   for (int i = 0; i < param.second; ++i)
01048       param.first->Increment();
01049 }
01050 
01051 // Tests that the mutex only lets one thread at a time to lock it.
01052 TEST(MutexTest, OnlyOneThreadCanLockAtATime) {
01053   Mutex mutex;
01054   AtomicCounterWithMutex locked_counter(&mutex);
01055 
01056   typedef ThreadWithParam<pair<AtomicCounterWithMutex*, int> > ThreadType;
01057   const int kCycleCount = 20;
01058   const int kThreadCount = 7;
01059   scoped_ptr<ThreadType> counting_threads[kThreadCount];
01060   Notification threads_can_start;
01061   // Creates and runs kThreadCount threads that increment locked_counter
01062   // kCycleCount times each.
01063   for (int i = 0; i < kThreadCount; ++i) {
01064     counting_threads[i].reset(new ThreadType(&CountingThreadFunc,
01065                                              make_pair(&locked_counter,
01066                                                        kCycleCount),
01067                                              &threads_can_start));
01068   }
01069   threads_can_start.Notify();
01070   for (int i = 0; i < kThreadCount; ++i)
01071     counting_threads[i]->Join();
01072 
01073   // If the mutex lets more than one thread to increment the counter at a
01074   // time, they are likely to encounter a race condition and have some
01075   // increments overwritten, resulting in the lower then expected counter
01076   // value.
01077   EXPECT_EQ(kCycleCount * kThreadCount, locked_counter.value());
01078 }
