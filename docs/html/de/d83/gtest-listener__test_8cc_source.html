00180 void VerifyResults(const std::vector<String>& data,
00181                    const char* const* expected_data,
00182                    int expected_data_size) {
00183   const int actual_size = data.size();
00184   // If the following assertion fails, a new entry will be appended to
00185   // data.  Hence we save data.size() first.
00186   EXPECT_EQ(expected_data_size, actual_size);
00187 
00188   // Compares the common prefix.
00189   const int shorter_size = expected_data_size <= actual_size ?
00190       expected_data_size : actual_size;
00191   int i = 0;
00192   for (; i < shorter_size; ++i) {
00193     ASSERT_STREQ(expected_data[i], data[i].c_str())
00194         << "at position " << i;
00195   }
00196 
00197   // Prints extra elements in the actual data.
00198   for (; i < actual_size; ++i) {
00199     printf("  Actual event #%d: %s\n", i, data[i].c_str());
00200   }
00201 }
