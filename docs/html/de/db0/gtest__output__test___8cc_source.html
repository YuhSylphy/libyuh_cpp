00234 // Tests that SCOPED_TRACE()s can be used concurrently from multiple
00235 // threads.  Namely, an assertion should be affected by
00236 // SCOPED_TRACE()s in its own thread only.

00238 // Here's the sequence of actions that happen in the test:
00239 //
00240 //   Thread A (main)                | Thread B (spawned)
00241 //   ===============================|================================
00242 //   spawns thread B                |
00243 //   -------------------------------+--------------------------------
00244 //   waits for n1                   | SCOPED_TRACE("Trace B");
00245 //                                  | generates failure #1
00246 //                                  | notifies n1
00247 //   -------------------------------+--------------------------------
00248 //   SCOPED_TRACE("Trace A");       | waits for n2
00249 //   generates failure #2           |
00250 //   notifies n2                    |
00251 //   -------------------------------|--------------------------------
00252 //   waits for n3                   | generates failure #3
00253 //                                  | trace B dies
00254 //                                  | generates failure #4
00255 //                                  | notifies n3
00256 //   -------------------------------|--------------------------------
00257 //   generates failure #5           | finishes
00258 //   trace A dies                   |
00259 //   generates failure #6           |
00260 //   -------------------------------|--------------------------------
00261 //   waits for thread B to finish   |
00262 
00263 struct CheckPoints {
00264   Notification n1;
00265   Notification n2;
00266   Notification n3;
00267 };
00268 
00269 static void ThreadWithScopedTrace(CheckPoints* check_points) {
00270   {
00271     SCOPED_TRACE("Trace B");
00272     ADD_FAILURE()
00273         << "Expected failure #1 (in thread B, only trace B alive).";
00274     check_points->n1.Notify();
00275     check_points->n2.WaitForNotification();
00276 
00277     ADD_FAILURE()
00278         << "Expected failure #3 (in thread B, trace A & B both alive).";
00279   }  // Trace B dies here.
00280   ADD_FAILURE()
00281       << "Expected failure #4 (in thread B, only trace A alive).";
00282   check_points->n3.Notify();
00283 }

00285 TEST(SCOPED_TRACETest, WorksConcurrently) {
00286   printf("(expecting 6 failures)\n");
00287 
00288   CheckPoints check_points;
00289   ThreadWithParam<CheckPoints*> thread(&ThreadWithScopedTrace,
00290                                        &check_points,

00292   check_points.n1.WaitForNotification();
00293 
00294   {
00295     SCOPED_TRACE("Trace A");
00296     ADD_FAILURE()
00297         << "Expected failure #2 (in thread A, trace A & B both alive).";

00977 // The main function.
00978 //
00979 // The idea is to use Google Test to run all the tests we have defined (some
00980 // of them are intended to fail), and then compare the test results
00981 // with the "golden" file.
00982 int main(int argc, char **argv) {
00983   testing::GTEST_FLAG(print_time) = false;
00984 
00985   // We just run the tests, knowing some of them are intended to fail.
00986   // We will use a separate Python script to compare the output of
00987   // this program with the golden file.
00988 
00989   // It's hard to test InitGoogleTest() directly, as it has many
00990   // global side effects.  The following line serves as a sanity test
00991   // for it.
00992   testing::InitGoogleTest(&argc, argv);
00993   if (argc >= 2 &&
00994       String(argv[1]) == "--gtest_internal_skip_environment_and_ad_hoc_tests")
00995     GTEST_FLAG(internal_skip_environment_and_ad_hoc_tests) = true;
00996 
00997 #if GTEST_HAS_DEATH_TEST
00998   if (testing::internal::GTEST_FLAG(internal_run_death_test) != "") {
00999     // Skip the usual output capturing if we're running as the child
01000     // process of an threadsafe-style death test.
01001 # if GTEST_OS_WINDOWS
01002     posix::FReopen("nul:", "w", stdout);
01003 # else
01004     posix::FReopen("/dev/null", "w", stdout);
01005 # endif  // GTEST_OS_WINDOWS
01006     return RUN_ALL_TESTS();
01007   }
01008 #endif  // GTEST_HAS_DEATH_TEST
01009 
01010   if (GTEST_FLAG(internal_skip_environment_and_ad_hoc_tests))
01011     return RUN_ALL_TESTS();
01012 
01013   // Registers two global test environments.
01014   // The golden file verifies that they are set up in the order they
01015   // are registered, and torn down in the reverse order.
01016   testing::AddGlobalTestEnvironment(new FooEnvironment);
01017   testing::AddGlobalTestEnvironment(new BarEnvironment);
01018 
01019   return RunAllTests();
01020 }
