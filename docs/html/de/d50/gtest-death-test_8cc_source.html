00208 // Routine for aborting the program which is safe to call from an
00209 // exec-style death test child process, in which case the error
00210 // message is propagated back to the parent process.  Otherwise, the
00211 // message is simply printed to stderr.  In either case, the program
00212 // then exits with status 1.
00213 void DeathTestAbort(const String& message) {
00214   // On a POSIX system, this function may be called from a threadsafe-style
00215   // death test child process, which operates on a very small stack.  Use
00216   // the heap for any additional non-minuscule memory requirements.
00217   const InternalRunDeathTestFlag* const flag =
00218       GetUnitTestImpl()->internal_run_death_test_flag();
00219   if (flag != NULL) {
00220     FILE* parent = posix::FDOpen(flag->write_fd(), "w");
00221     fputc(kDeathTestInternalError, parent);
00222     fprintf(parent, "%s", message.c_str());
00223     fflush(parent);
00224     _exit(1);
00225   } else {
00226     fprintf(stderr, "%s", message.c_str());
00227     fflush(stderr);
00228     posix::Abort();
00229   }

00381 // Called in the parent process only. Reads the result code of the death
00382 // test child process via a pipe, interprets it to set the outcome_
00383 // member, and closes read_fd_.  Outputs diagnostics and terminates in
00384 // case of unexpected codes.
00385 void DeathTestImpl::ReadAndInterpretStatusByte() {
00386   char flag;
00387   int bytes_read;
00388 
00389   // The read() here blocks until data is available (signifying the
00390   // failure of the death test) or until the pipe is closed (signifying
00391   // its success), so it's okay to call this in the parent before
00392   // the child process has exited.
00393   do {
00394     bytes_read = posix::Read(read_fd(), &flag, 1);
00395   } while (bytes_read == -1 && errno == EINTR);
00396 
00397   if (bytes_read == 0) {
00398     set_outcome(DIED);
00399   } else if (bytes_read == 1) {
00400     switch (flag) {
00401       case kDeathTestReturned:
00402         set_outcome(RETURNED);
00403         break;
00404       case kDeathTestThrew:
00405         set_outcome(THREW);
00406         break;
00407       case kDeathTestLived:
00408         set_outcome(LIVED);
00409         break;
00410       case kDeathTestInternalError:
00411         FailFromInternalError(read_fd());  // Does not return.
00412         break;
00413       default:
00414         GTEST_LOG_(FATAL) << "Death test child process reported "
00415                           << "unexpected status byte ("
00416                           << static_cast<unsigned int>(flag) << ")";
00417     }
00418   } else {
00419     GTEST_LOG_(FATAL) << "Read from death test child process failed: "
00420                       << GetLastErrnoDescription();
00421   }
00422   GTEST_DEATH_TEST_CHECK_SYSCALL_(posix::Close(read_fd()));
00423   set_read_fd(-1);
00424 }

00426 // Signals that the death test code which should have exited, didn't.
00427 // Should be called only in a death test child process.
00428 // Writes a status byte to the child's status file descriptor, then
00429 // calls _exit(1).
00430 void DeathTestImpl::Abort(AbortReason reason) {
00431   // The parent process considers the death test to be a failure if
00432   // it finds any data in our pipe.  So, here we write a single flag byte
00433   // to the pipe, then exit.
00434   const char status_ch =
00435       reason == TEST_DID_NOT_DIE ? kDeathTestLived :
00436       reason == TEST_THREW_EXCEPTION ? kDeathTestThrew : kDeathTestReturned;
00437 
00438   GTEST_DEATH_TEST_CHECK_SYSCALL_(posix::Write(write_fd(), &status_ch, 1));
00439   // We are leaking the descriptor here because on some platforms (i.e.,
00440   // when built as Windows DLL), destructors of global objects will still
00441   // run after calling _exit(). On such systems, write_fd_ will be
00442   // indirectly closed from the destructor of UnitTestImpl, causing double
00443   // close if it is also closed here. On debug configurations, double close
00444   // may assert. As there are no in-process buffers to flush here, we are
00445   // relying on the OS to close the descriptor after the process terminates
00446   // when the destructors are not run.
00447   _exit(1);  // Exits w/o any normal exit hooks (we were supposed to crash)
00448 }

00468 // Assesses the success or failure of a death test, using both private
00469 // members which have previously been set, and one argument:
00470 //
00471 // Private data members:
00472 //   outcome:  An enumeration describing how the death test
00473 //             concluded: DIED, LIVED, THREW, or RETURNED.  The death test
00474 //             fails in the latter three cases.
00475 //   status:   The exit status of the child process. On *nix, it is in the
00476 //             in the format specified by wait(2). On Windows, this is the
00477 //             value supplied to the ExitProcess() API or a numeric code
00478 //             of the exception that terminated the program.
00479 //   regex:    A regular expression object to be applied to
00480 //             the test's captured standard error output; the death test
00481 //             fails if it does not match.
00482 //
00483 // Argument:
00484 //   status_ok: true if exit_status is acceptable in the context of
00485 //              this particular death test, which fails if it is false
00486 //
00487 // Returns true iff all of the above conditions are met.  Otherwise, the
00488 // first failing condition, in the order given above, is the one that is
00489 // reported. Also sets the last death test message string.
00490 bool DeathTestImpl::Passed(bool status_ok) {
00491   if (!spawned())
00492     return false;
00493 
00494   const String error_message = GetCapturedStderr();
00495 
00496   bool success = false;
00497   Message buffer;
00498 
00499   buffer << "Death test: " << statement() << "\n";
00500   switch (outcome()) {
00501     case LIVED:
00502       buffer << "    Result: failed to die.\n"
00503              << " Error msg:\n" << FormatDeathTestOutput(error_message);
00504       break;
00505     case THREW:
00506       buffer << "    Result: threw an exception.\n"
00507              << " Error msg:\n" << FormatDeathTestOutput(error_message);
00508       break;
00509     case RETURNED:
00510       buffer << "    Result: illegal return in test statement.\n"
00511              << " Error msg:\n" << FormatDeathTestOutput(error_message);
00512       break;
00513     case DIED:
00514       if (status_ok) {
00515         const bool matched = RE::PartialMatch(error_message.c_str(), *regex());
00516         if (matched) {
00517           success = true;
00518         } else {
00519           buffer << "    Result: died but not with expected error.\n"
00520                  << "  Expected: " << regex()->pattern() << "\n"
00521                  << "Actual msg:\n" << FormatDeathTestOutput(error_message);
00522         }
00523       } else {
00524         buffer << "    Result: died but not with expected exit code:\n"
00525                << "            " << ExitSummary(status()) << "\n"
00526                << "Actual msg:\n" << FormatDeathTestOutput(error_message);
00527       }
00528       break;
00529     case IN_PROGRESS:
00530     default:
00531       GTEST_LOG_(FATAL)
00532           << "DeathTest::Passed somehow called before conclusion of test";
00533   }
00534 
00535   DeathTest::set_last_death_test_message(buffer.GetString());
00536   return success;
00537 }

00540 // WindowsDeathTest implements death tests on Windows. Due to the
00541 // specifics of starting new processes on Windows, death tests there are
00542 // always threadsafe, and Google Test considers the
00543 // --gtest_death_test_style=fast setting to be equivalent to
00544 // --gtest_death_test_style=threadsafe there.
00545 //
00546 // A few implementation notes:  Like the Linux version, the Windows
00547 // implementation uses pipes for child-to-parent communication. But due to
00548 // the specifics of pipes on Windows, some extra steps are required:
00549 //
00550 // 1. The parent creates a communication pipe and stores handles to both
00551 //    ends of it.
00552 // 2. The parent starts the child and provides it with the information
00553 //    necessary to acquire the handle to the write end of the pipe.
00554 // 3. The child acquires the write end of the pipe and signals the parent
00555 //    using a Windows event.
00556 // 4. Now the parent can release the write end of the pipe on its side. If
00557 //    this is done before step 3, the object's reference count goes down to
00558 //    0 and it is destroyed, preventing the child from acquiring it. The
00559 //    parent now has to release it, or read operations on the read end of
00560 //    the pipe will not return when the child terminates.
00561 // 5. The parent reads child's output through the pipe (outcome code and
00562 //    any possible error messages) from the pipe, and its stderr and then
00563 //    determines whether to fail the test.
00564 //
00565 // Note: to distinguish Win32 API calls from the local method and function
00566 // calls, the former are explicitly resolved in the global namespace.
00567 //
00568 class WindowsDeathTest : public DeathTestImpl {
00569  public:
00570   WindowsDeathTest(const char* a_statement,
00571                    const RE* a_regex,
00572                    const char* file,
00573                    int line)
00574       : DeathTestImpl(a_statement, a_regex), file_(file), line_(line) {}
00575 
00576   // All of these virtual functions are inherited from DeathTest.
00577   virtual int Wait();
00578   virtual TestRole AssumeRole();
00579 
00580  private:
00581   // The name of the file in which the death test is located.
00582   const char* const file_;
00583   // The line number on which the death test is located.

00590   // acquired the handle to the write end of the pipe. After seeing this

00596 // Waits for the child in a death test to exit, returning its exit
00597 // status, or 0 if no child process exists.  As a side effect, sets the
00598 // outcome data member.
00599 int WindowsDeathTest::Wait() {
00600   if (!spawned())
00601     return 0;
00602 
00603   // Wait until the child either signals that it has acquired the write end
00604   // of the pipe or it dies.
00605   const HANDLE wait_handles[2] = { child_handle_.Get(), event_handle_.Get() };
00606   switch (::WaitForMultipleObjects(2,
00607                                    wait_handles,
00608                                    FALSE,  // Waits for any of the handles.
00609                                    INFINITE)) {
00610     case WAIT_OBJECT_0:
00611     case WAIT_OBJECT_0 + 1:
00612       break;
00613     default:
00614       GTEST_DEATH_TEST_CHECK_(false);  // Should not get here.
00615   }
00616 
00617   // The child has acquired the write end of the pipe or exited.
00618   // We release the handle on our side and continue.
00619   write_handle_.Reset();
00620   event_handle_.Reset();
00621 
00622   ReadAndInterpretStatusByte();
00623 
00624   // Waits for the child process to exit if it haven't already. This
00625   // returns immediately if the child has already exited, regardless of
00626   // whether previous calls to WaitForMultipleObjects synchronized on this
00627   // handle or not.
00628   GTEST_DEATH_TEST_CHECK_(
00629       WAIT_OBJECT_0 == ::WaitForSingleObject(child_handle_.Get(),
00630                                              INFINITE));
00631   DWORD status_code;
00632   GTEST_DEATH_TEST_CHECK_(
00633       ::GetExitCodeProcess(child_handle_.Get(), &status_code) != FALSE);
00634   child_handle_.Reset();
00635   set_status(static_cast<int>(status_code));
00636   return status();
00637 }

00784 // The AssumeRole process for a fork-and-run death test.  It implements a
00785 // straightforward fork, with a simple pipe to transmit the status byte.
00786 DeathTest::TestRole NoExecDeathTest::AssumeRole() {
00787   const size_t thread_count = GetThreadCount();
00788   if (thread_count != 1) {
00789     GTEST_LOG_(WARNING) << DeathTestThreadWarning(thread_count);
00790   }
00791 
00792   int pipe_fd[2];
00793   GTEST_DEATH_TEST_CHECK_(pipe(pipe_fd) != -1);
00794 
00795   DeathTest::set_last_death_test_message("");
00796   CaptureStderr();
00797   // When we fork the process below, the log file buffers are copied, but the
00798   // file descriptors are shared.  We flush all log files here so that closing
00799   // the file descriptors in the child process doesn't throw off the
00800   // synchronization between descriptors and buffers in the parent process.
00801   // This is as close to the fork as possible to avoid a race condition in case
00802   // there are multiple threads running before the death test, and another
00803   // thread writes to the log file.
00804   FlushInfoLog();
00805 
00806   const pid_t child_pid = fork();
00807   GTEST_DEATH_TEST_CHECK_(child_pid != -1);
00808   set_child_pid(child_pid);
00809   if (child_pid == 0) {
00810     GTEST_DEATH_TEST_CHECK_SYSCALL_(close(pipe_fd[0]));
00811     set_write_fd(pipe_fd[1]);
00812     // Redirects all logging to stderr in the child process to prevent
00813     // concurrent writes to the log files.  We capture stderr in the parent
00814     // process and append the child process' output to a log.
00815     LogToStderr();
00816     // Event forwarding to the listeners of event listener API mush be shut
00817     // down in death test subprocesses.
00818     GetUnitTestImpl()->listeners()->SuppressEventForwarding();
00819     return EXECUTE_TEST;
00820   } else {
00821     GTEST_DEATH_TEST_CHECK_SYSCALL_(close(pipe_fd[1]));
00822     set_read_fd(pipe_fd[0]);
00823     set_spawned(true);
00824     return OVERSEE_TEST;
00825   }
00826 }

00897 // The main function for a threadsafe-style death test child process.
00898 // This function is called in a clone()-ed process and thus must avoid
00899 // any potentially unsafe operations like malloc or libc functions.
00900 static int ExecDeathTestChildMain(void* child_arg) {
00901   ExecDeathTestArgs* const args = static_cast<ExecDeathTestArgs*>(child_arg);
00902   GTEST_DEATH_TEST_CHECK_SYSCALL_(close(args->close_fd));
00903 
00904   // We need to execute the test program in the same environment where
00905   // it was originally invoked.  Therefore we change to the original
00906   // working directory first.
00907   const char* const original_dir =
00908       UnitTest::GetInstance()->original_working_dir();
00909   // We can safely call chdir() as it's a direct system call.
00910   if (chdir(original_dir) != 0) {
00911     DeathTestAbort(String::Format("chdir(\"%s\") failed: %s",
00912                                   original_dir,
00913                                   GetLastErrnoDescription().c_str()));
00914     return EXIT_FAILURE;
00915   }
00916 
00917   // We can safely call execve() as it's a direct system call.  We
00918   // cannot use execvp() as it's a libc function and thus potentially
00919   // unsafe.  Since execve() doesn't search the PATH, the user must
00920   // invoke the test program via a valid path that contains at least
00921   // one path separator.
00922   execve(args->argv[0], args->argv, GetEnviron());
00923   DeathTestAbort(String::Format("execve(%s, ...) in %s failed: %s",
00924                                 args->argv[0],
00925                                 original_dir,
00926                                 GetLastErrnoDescription().c_str()));
00927   return EXIT_FAILURE;
00928 }

00930 // Two utility routines that together determine the direction the stack
00931 // grows.
00932 // This could be accomplished more elegantly by a single recursive
00933 // function, but we want to guard against the unlikely possibility of
00934 // a smart compiler optimizing the recursion away.
00935 //
00936 // GTEST_NO_INLINE_ is required to prevent GCC 4.6 from inlining
00937 // StackLowerThanAddress into StackGrowsDown, which then doesn't give
00938 // correct answer.
00939 bool StackLowerThanAddress(const void* ptr) GTEST_NO_INLINE_;
00940 bool StackLowerThanAddress(const void* ptr) {
00941   int dummy;
00942   return &dummy < ptr;
00943 }
00944 
00945 bool StackGrowsDown() {
00946   int dummy;
00947   return StackLowerThanAddress(&dummy);
00948 }

00950 // A threadsafe implementation of fork(2) for threadsafe-style death tests
00951 // that uses clone(2).  It dies with an error message if anything goes
00952 // wrong.
00953 static pid_t ExecDeathTestFork(char* const* argv, int close_fd) {
00954   ExecDeathTestArgs args = { argv, close_fd };
00955   pid_t child_pid = -1;
00956 
00957 #  if GTEST_HAS_CLONE
00958   const bool use_fork = GTEST_FLAG(death_test_use_fork);
00959 
00960   if (!use_fork) {
00961     static const bool stack_grows_down = StackGrowsDown();
00962     const size_t stack_size = getpagesize();
00963     // MMAP_ANONYMOUS is not defined on Mac, so we use MAP_ANON instead.
00964     void* const stack = mmap(NULL, stack_size, PROT_READ | PROT_WRITE,
00965                              MAP_ANON | MAP_PRIVATE, -1, 0);
00966     GTEST_DEATH_TEST_CHECK_(stack != MAP_FAILED);
00967     void* const stack_top =
00968         static_cast<char*>(stack) + (stack_grows_down ? stack_size : 0);
00969 
00970     child_pid = clone(&ExecDeathTestChildMain, stack_top, SIGCHLD, &args);
00971 
00972     GTEST_DEATH_TEST_CHECK_(munmap(stack, stack_size) != -1);
00973   }
00974 #  else
00975   const bool use_fork = true;
00976 #  endif  // GTEST_HAS_CLONE
00977 
00978   if (use_fork && (child_pid = fork()) == 0) {
00979       ExecDeathTestChildMain(&args);
00980       _exit(0);
00981   }
00982 
00983   GTEST_DEATH_TEST_CHECK_(child_pid != -1);
00984   return child_pid;
00985 }

01095 // Splits a given string on a given delimiter, populating a given
01096 // vector with the fields.  GTEST_HAS_DEATH_TEST implies that we have
01097 // ::std::string, so we can use it here.
01098 static void SplitString(const ::std::string& str, char delimiter,
01099                         ::std::vector< ::std::string>* dest) {
01100   ::std::vector< ::std::string> parsed;
01101   ::std::string::size_type pos = 0;
01102   while (::testing::internal::AlwaysTrue()) {
01103     const ::std::string::size_type colon = str.find(delimiter, pos);
01104     if (colon == ::std::string::npos) {
01105       parsed.push_back(str.substr(pos));
01106       break;
01107     } else {
01108       parsed.push_back(str.substr(pos, colon - pos));
01109       pos = colon + 1;
01110     }
01111   }
01112   dest->swap(parsed);
01113 }

01116 // Recreates the pipe and event handles from the provided parameters,
01117 // signals the event, and returns a file descriptor wrapped around the pipe
01118 // handle. This function is called in the child process only.
01119 int GetStatusFileDescriptor(unsigned int parent_process_id,
01120                             size_t write_handle_as_size_t,
01121                             size_t event_handle_as_size_t) {
01122   AutoHandle parent_process_handle(::OpenProcess(PROCESS_DUP_HANDLE,
01123                                                    FALSE,  // Non-inheritable.
01124                                                    parent_process_id));
01125   if (parent_process_handle.Get() == INVALID_HANDLE_VALUE) {
01126     DeathTestAbort(String::Format("Unable to open parent process %u",
01127                                   parent_process_id));
01128   }
01129 
01130   // TODO(vladl@google.com): Replace the following check with a
01131   // compile-time assertion when available.
01132   GTEST_CHECK_(sizeof(HANDLE) <= sizeof(size_t));
01133 
01134   const HANDLE write_handle =
01135       reinterpret_cast<HANDLE>(write_handle_as_size_t);
01136   HANDLE dup_write_handle;
01137 
01138   // The newly initialized handle is accessible only in in the parent
01139   // process. To obtain one accessible within the child, we need to use
01140   // DuplicateHandle.
01141   if (!::DuplicateHandle(parent_process_handle.Get(), write_handle,
01142                          ::GetCurrentProcess(), &dup_write_handle,
01143                          0x0,    // Requested privileges ignored since
01144                                  // DUPLICATE_SAME_ACCESS is used.
01145                          FALSE,  // Request non-inheritable handler.
01146                          DUPLICATE_SAME_ACCESS)) {
01147     DeathTestAbort(String::Format(
01148         "Unable to duplicate the pipe handle %Iu from the parent process %u",
01149         write_handle_as_size_t, parent_process_id));
01150   }
01151 
01152   const HANDLE event_handle = reinterpret_cast<HANDLE>(event_handle_as_size_t);
01153   HANDLE dup_event_handle;
01154 
01155   if (!::DuplicateHandle(parent_process_handle.Get(), event_handle,
01156                          ::GetCurrentProcess(), &dup_event_handle,
