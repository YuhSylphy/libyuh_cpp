00049 // Tests that death tests report thrown exceptions as failures and that the
00050 // exceptions do not escape death test macros.
00051 TEST(CxxExceptionDeathTest, ExceptionIsFailure) {
00052   try {
00053     EXPECT_NONFATAL_FAILURE(EXPECT_DEATH(throw 1, ""), "threw an exception");
