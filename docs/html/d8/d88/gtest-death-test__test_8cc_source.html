00106 void DieWithMessage(const ::std::string& message) {
00107   fprintf(stderr, "%s", message.c_str());
00108   fflush(stderr);  // Make sure the text is printed before the process exits.
00109 
00110   // We call _exit() instead of exit(), as the former is a direct
00111   // system call and thus safer in the presence of threads.  exit()
00112   // will invoke user-defined exit-hooks, which may do dangerous
00113   // things that conflict with death tests.
00114   //
00115   // Some compilers can recognize that _exit() never returns and issue the
00116   // 'unreachable code' warning for code following this function, unless
00117   // fooled by a fake condition.
00118   if (AlwaysTrue())
00119     _exit(1);
00120 }

00223 // Returns the exit status of a process that calls _exit(2) with a
00224 // given exit code.  This is a helper function for the
00225 // ExitStatusPredicateTest test suite.
00226 static int NormalExitStatus(int exit_code) {
00227   pid_t child_pid = fork();
00228   if (child_pid == 0) {
00229     _exit(exit_code);
00230   }
00231   int status;
00232   waitpid(child_pid, &status, 0);
00233   return status;
00234 }

00236 // Returns the exit status of a process that raises a given signal.
00237 // If the signal does not cause the process to die, then it returns
00238 // instead the exit status of a process that exits normally with exit
00239 // code 1.  This is a helper function for the ExitStatusPredicateTest
00240 // test suite.
00241 static int KilledExitStatus(int signum) {
00242   pid_t child_pid = fork();
00243   if (child_pid == 0) {
00244     raise(signum);
00245     _exit(1);
00246   }
00247   int status;
00248   waitpid(child_pid, &status, 0);
00249   return status;
00250 }

00814 // A DeathTest implementation useful in testing.  It returns values set
00815 // at its creation from its various inherited DeathTest methods, and
00816 // reports calls to those methods to its parent MockDeathTestFactory
00817 // object.
00818 class MockDeathTest : public DeathTest {
00819  public:
00820   MockDeathTest(MockDeathTestFactory *parent,
00821                 TestRole role, int status, bool passed) :
00822       parent_(parent), role_(role), status_(status), passed_(passed) {
00823   }
00824   virtual ~MockDeathTest() {
00825     parent_->test_deleted_ = true;
00826   }
00827   virtual TestRole AssumeRole() {
00828     ++parent_->assume_role_calls_;
00829     return role_;
00830   }
00831   virtual int Wait() {
00832     ++parent_->wait_calls_;
00833     return status_;
00834   }
00835   virtual bool Passed(bool exit_status_ok) {
00836     parent_->passed_args_.push_back(exit_status_ok);
00837     return passed_;
00838   }
00839   virtual void Abort(AbortReason reason) {
00840     parent_->abort_args_.push_back(reason);
00841   }
00842  private:
00843   MockDeathTestFactory* const parent_;
00844   const TestRole role_;
00845   const int status_;
00846   const bool passed_;
00847 };
